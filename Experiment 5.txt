#include <stdio.h>
int factRec(int n) {
    if (n == 0)
        return 1;
    return n * factRec(n - 1);
}
int factIter(int n) {
    int fact = 1;
    for (int i = 1; i <= n; i++) {
        fact *= i;
    }
    return fact;
}
int main() {
    int n;
    printf("Enter a number: ");
    scanf("%d", &n);
    printf("Factorial using recursion = %d\n", factRec(n));
    printf("Factorial using iteration = %d\n", factIter(n));

    return 0;
}
